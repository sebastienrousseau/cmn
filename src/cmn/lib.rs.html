7//! # Common (CMN)
8//!
9//! 121 mathematical and cryptographic constants for Rust.
10//! Zero runtime cost. `no_std` compatible.
11//!
12//! ## Why CMN?
13//!
14//! `std::f64::consts` provides 11 mathematical constants. CMN
15//! extends that with 44 physical, cryptographic, and series
16//! constants — plus a runtime lookup API, 14 utility macros,
17//! and a built-in word-list module. Every constant resolves at
18//! compile time. Works in `no_std` environments.
19//!
20//! ## Modules
21//!
22//! - **[`constants`]** — 121 `const` values (PI, Avogadro,
23//!   Planck, etc.). With `std`: runtime `Constants` lookup API
24//!   + `ConstantValue` typed enum.
25//! - **[`macros`]** — 14 utility macros for min/max,
26//!   range-checks, collections, and parsing.
27//! - With `std` feature: **[`words`]** module, **[`datetime`]**
28//!   utilities, and **`Common`** JSON bridge.
29//!
30//! ## Quick Start
31//!
32//! ```rust
33//! // Compile-time constants — always available, even in no_std
34//! use cmn::constants::{PI, EULER, SPEED_OF_LIGHT};
35//!
36//! assert_eq!(PI, core::f64::consts::PI);
37//! ```
38//!
39//! ```rust
40//! // Runtime lookup — requires the `std` feature (default)
41//! use cmn::Constants;
42//! use cmn::Words;
43//! use cmn::words::WORD_LIST;
44//!
45//! let constants = Constants::new();
46//! let euler = constants.constant("EULER").unwrap();
47//! assert_eq!(euler.name, "EULER");
48//!
49//! let words = Words::default();
50//! let list = words.words_list();
51//! assert_eq!(list.len(), WORD_LIST.len());
52//! assert_eq!(list[0], "aboard");
53//! ```
54//!
55//! ## Feature Flags
56//!
57//! | Feature | Default | Enables |
58//! |---------|---------|---------|
59//! | `std`   | Yes     | `Constants` struct, `Words`, `Common`, serde support |
60//!
61//! For `no_std`, disable default features:
62//! ```toml
63//! [dependencies]
64//! cmn = { version = "0.0.6", default-features = false }
65//! ```

110/// The `Common` structure provides a central location to store
111/// data that is commonly used throughout the library. Requires
112/// the `std` feature for serde JSON support.
113///
114/// # Fields
115///
116/// * `constants`: A reference to the `Constants` structure.
117/// * `words`: A reference to the `Words` structure.
118#[cfg(feature = "std")]
119#[derive(Clone, Serialize, Deserialize, Debug)]
120pub struct Common {
121    #[serde(flatten)]
122    fields: serde_json::Value,
123}
124
125#[cfg(feature = "std")]
126impl Common {
127    /// Creates a new instance of the `Common` structure.
128    pub fn new() -> Self {
129        Self {
130            fields: serde_json::Value::Null,
131        }
132    }
133    /// Returns the `Constants` instance.
134    pub fn constants(&self) -> Constants {
135        Constants {
136            ..Default::default()
137        }
138    }
139    /// Returns a new instance of the `Words` structure.
140    ///
141    /// Gracefully handles malformed JSON — returns an empty
142    /// `Words` if the `words` field is missing, not an array,
143    /// or contains non-string elements.
144    pub fn words(&self) -> Words {
145        let words_data: Vec<String> = self
146            .fields
147            .get("words")
148            .and_then(|v| v.as_array())
149            .map(|arr| {
150                arr.iter()
151                    .filter_map(|v| v.as_str().map(|s| s.to_string()))
152                    .collect()
153            })
154            .unwrap_or_default();
155
156        Words {
157            words: words_data.into_iter().collect(),
158        }
159    }
160
161    /// Parses a string of JSON data and returns a new
162    /// instance of the `Common` structure.
163    pub fn parse(input: &str) -> Result<Self, serde_json::Error> {
164        let common = serde_json::from_str::<Common>(input)?;
165        let is_empty = common.fields.is_null()
166            || common
167                .fields
168                .as_object()
169                .is_some_and(|obj| obj.is_empty());
170        if is_empty {
171            Ok(Common::default())
172        } else {
173            Ok(common)
174        }
175    }
176}
177
178#[cfg(feature = "std")]
179impl Default for Common {
180    fn default() -> Self {
181        Self::new()
182    }
183}
